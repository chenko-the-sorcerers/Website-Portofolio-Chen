01

Setup & Dataset CIFAR-10

Persiapan environment dan loading dataset — langkah pertama setiap project DL

Kita akan membangun dan melatih CNN dari scratch menggunakan PyTorch pada dataset CIFAR-10 — 60,000 gambar 32×32 dari 10 kelas: pesawat, mobil, burung, kucing, rusa, anjing, katak, kuda, kapal, truk.
python 01_setup.py 
import torch
import torch.nn as nn
import torch.optim as optim
import torchvision
import torchvision.transforms as transforms
import matplotlib.pyplot as plt

# ── Device ─────────────────────────────────────────────
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
print(f'Using device: {device}')

# ── Data transforms ────────────────────────────────────
transform_train = transforms.Compose([
    transforms.RandomHorizontalFlip(),     # augmentasi
    transforms.RandomCrop(32, padding=4),  # augmentasi
    transforms.ToTensor(),
    transforms.Normalize(
        mean=(0.4914, 0.4822, 0.4465),  # mean CIFAR-10
        std= (0.2023, 0.1994, 0.2010)   # std CIFAR-10
    )
])
transform_test = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize(
        mean=(0.4914, 0.4822, 0.4465),
        std= (0.2023, 0.1994, 0.2010)
    )
])

# ── Load dataset ───────────────────────────────────────
train_set = torchvision.datasets.CIFAR10(
    root='./data', train=True, download=True, transform=transform_train
)
test_set = torchvision.datasets.CIFAR10(
    root='./data', train=False, download=True, transform=transform_test
)

train_loader = torch.utils.data.DataLoader(train_set, batch_size=128, shuffle=True,  num_workers=2)
test_loader  = torch.utils.data.DataLoader(test_set,  batch_size=128, shuffle=False, num_workers=2)

CLASSES = ['pesawat', 'mobil', 'burung', 'kucing', 'rusa',
           'anjing',  'katak', 'kuda',  'kapal', 'truk']
💡
Normalisasi penting. Kita kurangi mean dan bagi dengan std dari dataset training. Ini membuat gradient lebih stabil dan training lebih cepat konvergen. Nilai mean/std CIFAR-10 sudah dihitung dan tersedia di dokumentasi torchvision.

Definisi Model CNN

Arsitektur sederhana tapi efektif — Conv → ReLU → Pool → FC → Softmax

Kita bangun CNN kecil dengan dua blok Conv+ReLU+Pool, diikuti Global Average Pooling dan satu FC layer. Target: ~75% akurasi di CIFAR-10 dengan <100K parameter.
python 02_model.py 
class SimpleCNN(nn.Module):
    def __init__(self, num_classes=10):
        super().__init__()

        # ── Feature extractor ──────────────────────────────
        self.features = nn.Sequential(
            # Block 1: 32×32×3 → 16×16×32
            nn.Conv2d(3,  32, kernel_size=3, padding=1),  # 896 params
            nn.BatchNorm2d(32),
            nn.ReLU(inplace=True),
            nn.Conv2d(32, 32, kernel_size=3, padding=1),  # 9,248 params
            nn.BatchNorm2d(32),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2),                              # 32×32 → 16×16
            nn.Dropout2d(0.1),

            # Block 2: 16×16×32 → 8×8×64
            nn.Conv2d(32, 64, kernel_size=3, padding=1),  # 18,496 params
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),
            nn.Conv2d(64, 64, kernel_size=3, padding=1),  # 36,928 params
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2),                              # 16×16 → 8×8
            nn.Dropout2d(0.2),
        )

        # ── Classifier ─────────────────────────────────────
        self.gap        = nn.AdaptiveAvgPool2d(1)   # Global Avg Pool → 1×1×64
        self.classifier = nn.Linear(64, num_classes) # 640 + 10 bias

    def forward(self, x):
        x = self.features(x)    # (B, 3, 32, 32) → (B, 64, 8, 8)
        x = self.gap(x)         # (B, 64, 8, 8)  → (B, 64, 1, 1)
        x = x.flatten(1)        # (B, 64, 1, 1)  → (B, 64)
        x = self.classifier(x)  # (B, 64)        → (B, 10) logits
        return x                # Softmax diterapkan di loss function

model = SimpleCNN().to(device)
total_params = sum(p.numel() for p in model.parameters() if p.requires_grad)
print(f'Total trainable params: {total_params:,}')
# → Total trainable params: 66,442

Training Loop

Forward pass → loss → backward → optimizer step — diulang ribuan kali

Training loop adalah inti dari setiap deep learning project. Setiap epoch, kita lewatkan seluruh training set, hitung loss, propagasikan gradient mundur, lalu update weight.
python 03_train.py 
# ── Loss & optimizer ───────────────────────────────────
criterion = nn.CrossEntropyLoss()            # Softmax sudah di dalam ini
optimizer = optim.AdamW(
    model.parameters(), lr=1e-3, weight_decay=1e-4
)
scheduler = optim.lr_scheduler.CosineAnnealingLR(
    optimizer, T_max=50
)

# ── Training loop ──────────────────────────────────────
def train_epoch(model, loader, criterion, optimizer):
    model.train()
    total_loss, correct, total = 0.0, 0, 0

    for images, labels in loader:
        images, labels = images.to(device), labels.to(device)

        optimizer.zero_grad()           # reset gradient lama
        outputs = model(images)           # forward pass
        loss    = criterion(outputs, labels)  # hitung loss
        loss.backward()                  # backprop
        optimizer.step()                 # update weights

        total_loss += loss.item() * images.size(0)
        _, predicted = outputs.max(1)
        correct += predicted.eq(labels).sum().item()
        total   += labels.size(0)

    return total_loss / total, 100.0 * correct / total

# ── Eval loop ──────────────────────────────────────────
def evaluate(model, loader, criterion):
    model.eval()
    total_loss, correct, total = 0.0, 0, 0
    with torch.no_grad():
        for images, labels in loader:
            images, labels = images.to(device), labels.to(device)
            outputs = model(images)
            loss    = criterion(outputs, labels)
            total_loss += loss.item() * images.size(0)
            _, predicted = outputs.max(1)
            correct += predicted.eq(labels).sum().item()
            total   += labels.size(0)
    return total_loss / total, 100.0 * correct / total

# ── Run training ───────────────────────────────────────
EPOCHS = 50
history = {'train_loss': [], 'val_loss': [], 'val_acc': []}

for epoch in range(1, EPOCHS + 1):
    tr_loss, tr_acc = train_epoch(model, train_loader, criterion, optimizer)
    vl_loss, vl_acc = evaluate(model, test_loader, criterion)
    scheduler.step()

    history['train_loss'].append(tr_loss)
    history['val_loss'].append(vl_loss)
    history['val_acc'].append(vl_acc)

    if epoch % 5 == 0:
        print(f'Epoch {epoch:2d} | '
              f'Train loss: {tr_loss:.3f} acc: {tr_acc:.1f}% | '
              f'Val loss: {vl_loss:.3f} acc: {vl_acc:.1f}%')

Evaluasi & Visualisasi

Memahami hasil training — confusion matrix, per-class accuracy, learning curves

python 04_eval.py 
import numpy as np
from sklearn.metrics import confusion_matrix, classification_report
import seaborn as sns

# ── Collect all predictions ────────────────────────────
model.eval()
all_preds, all_labels = [], []

with torch.no_grad():
    for images, labels in test_loader:
        images = images.to(device)
        outputs = model(images)
        _, predicted = outputs.max(1)
        all_preds.extend(predicted.cpu().numpy())
        all_labels.extend(labels.numpy())

all_preds  = np.array(all_preds)
all_labels = np.array(all_labels)

# ── Per-class accuracy ─────────────────────────────────
print(classification_report(all_labels, all_preds, target_names=CLASSES))

# ── Confusion matrix ───────────────────────────────────
cm = confusion_matrix(all_labels, all_preds)
plt.figure(figsize=(10, 8))
sns.heatmap(cm, annot=True, fmt='d', cmap='Blues',
            xticklabels=CLASSES, yticklabels=CLASSES)
plt.title('Confusion Matrix — SimpleCNN CIFAR-10')
plt.ylabel('True Label'); plt.xlabel('Predicted Label')
plt.show()

# ── Learning curves ────────────────────────────────────
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 4))

ax1.plot(history['train_loss'], label='Train Loss', color='#2997ff')
ax1.plot(history['val_loss'],   label='Val Loss',   color='#ff3b30')
ax1.set_title('Loss Curve'); ax1.legend()

ax2.plot(history['val_acc'], label='Val Accuracy', color='#34c759')
ax2.axhline(75, linestyle='--', color='gray', label='Target 75%')
ax2.set_title('Validation Accuracy'); ax2.legend()

plt.tight_layout(); plt.show()

Cara Meningkatkan Akurasi

Dari 75% ke 90%+ — teknik-teknik yang benar-benar bekerja

Model kita mencapai ~75% akurasi. State-of-the-art CIFAR-10 mencapai >99%. Berikut langkah-langkah praktis yang bisa kamu coba, dari yang paling mudah ke yang paling complex:
1
Data Augmentation
Tambah RandomRotation, ColorJitter, CutOut. Efektif dan mudah.
✓ +2–5% akurasi
2
Lebih Banyak Layer
Tambah Conv block ke-3 dengan 128 filter. Lebih ekspresif.
✓ +3–7% akurasi
3
Learning Rate Warmup
Mulai dari LR kecil, naikan 5 epoch pertama, lalu cosine decay.
✓ Training lebih stabil
4
Transfer Learning
Gunakan ResNet-18 pre-trained ImageNet, fine-tune classifier saja.
✓ 90%+ dengan cepat
